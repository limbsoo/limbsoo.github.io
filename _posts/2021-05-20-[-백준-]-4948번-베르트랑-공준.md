```c++
#include <iostream>
#include <cmath>

using namespace std;

bool decimal(int n, int m)
{
    if (n < 2) // 1은 소수 X
    {
        return false;
    }

    int a = (int)sqrt(n); //약수는 본인의 루트값보다 클수 없으므로 범위를 줄임

    for (int i = 2; i <= a; i++) 
    {
        if (n % i == 0) 
        {
            return false;
        }
    }
    return true;
}

int main() 
{
    int n;
    

    while (1)
    {
        cin >> n;
        int cnt = 0;
        int m = 2 * n;

        if (!n)
        {
            break;
        }

        n++;

        for (; n <= m; n++) // n~2n까지
        {
           
            bool isPos = decimal(n,m);

            if (isPos)
            {
                cnt++;
            }
        }
        printf("%d\n", cnt);
    }

    return 0;
}
```
