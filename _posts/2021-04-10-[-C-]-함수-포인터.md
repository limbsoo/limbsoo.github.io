# 함수 포인터
___
<br>
- 특정 함수의 반환 자료형을 지정하는 방식으로 선언 가능

- 함수 포인터를 이용하면 형태가 같은 서로 다른 기능의 함수를 선택적 사용 가능
<br>

```c

int add(int a, int b) {

return a + b;

}

int sub(int a, int b) {

return a - b;

}

int main(void) {

//반환 자료형(*이름)(매개변수) = 함수명;
int(*fp)(int, int) = add;

printf("%d\n", fp(10, 3));

fp = sub;

printf("%d\n",fp(10,3));

system("pause");

return 0;

}

>> 10

      7
```
<br>


```c

int add(int a, int b) {

return a + b;

}

int(*process(char*a))(int, int) {

printf("%s\n", a);

return add;

}

int main(void) {

printf("%d\n", process("10과 20을 더한 값을 출력합니다")(10, 20));

system("pause");

return 0;

}

>> 10과 20을 더한 값을 출력합니다 

     30


```
